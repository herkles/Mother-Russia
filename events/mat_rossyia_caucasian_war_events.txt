					}
					if = {
						limit = {
							any_state = {
								state_region = s:STATE_ELIZAVETPOL
								owner ?= c:RUS
							}
						}
						random_state = {
							limit = {
								owner ?= c:RUS
								state_region = s:STATE_ELIZAVETPOL
							}
							set_state_owner = c:CAU
						}
					}
					if = {
						limit = {
							any_state = {
								state_region = s:STATE_KARS
								owner ?= c:RUS
							}
						}
						random_state = {
							limit = {
								owner ?= c:RUS
								state_region = s:STATE_KARS
							}
							set_state_owner = c:CAU
						}
					}
					activate_law = law_type:law_presidential_republic
					set_ruling_interest_groups = { ig_intelligentsia }
				}
			}
		}
	}
}
