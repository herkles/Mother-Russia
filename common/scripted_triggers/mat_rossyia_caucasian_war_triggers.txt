mat_rossyia_caucasian_war_should_be_involved_trigger = {
	custom_tooltip = {
		text = mat_rossyia_caucasian_war_should_be_involved_trigger
		OR = {
			c:RUS ?= this
			AND = {
				is_subject_of = c:RUS
				capital ?= {
					region = sr:region_caucasus
				}
			}
			any_scope_state = {
				region = sr:region_caucasus
				is_capital = yes
			}
			mat_rossiya_has_military_assistance_trigger = {
				country = c:RUS
			}
			mat_rossiya_has_rivalry_trigger = {
				country = c:RUS
			}
			mat_rossiya_has_embargo_trigger = {
				country = c:RUS
			}
			AND = {
				country_rank >= rank_value:great_power
				has_interest_marker_in_region = region_caucasus
			}
		}
	}
}

mat_rossyia_caucasian_war_interloper_trigger = {
	custom_tooltip = {
		text = mat_rossyia_caucasian_war_interloper_trigger
		mat_rossyia_caucasian_war_should_be_involved_trigger = no
		OR = {
			has_diplomatic_relevance = c:RUS
			any_scope_state = {
				region = sr:region_caucasus
			}
			any_subject_or_below = {
				capital = {
					region = sr:region_caucasus
				}
			}
			AND = {
				is_subject_of = c:RUS
				capital ?= {
					region != sr:region_caucasus
				}
			}
		}
	}
}
